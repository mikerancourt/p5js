let num;
let radius;
let spacing;
let quotientX;
let quotientY;

function setup() {
  createCanvas(windowWidth, windowHeight);
  background(0);

  num = 10;
  radius = (windowWidth + windowHeight) / num;
  spacing = num / 2;
  quotientX = windowWidth / num; 
  quotientY = windowHeight / num;
}

function draw() 
{
  background(0);
  
  let y = 0;
  while (y < num * quotientY - radius / 2) 
  {
    let x = 0;
    
    while (x < num * quotientX + radius / 2) 
    {
      fill(48);
      
      let d = dist(mouseX, mouseY, x + radius/2, y - radius/2);
      
      if (d < radius/2 ) 
      {
        fill(127);
        
        if (mouseIsPressed)
          {
            fill(255);
          }
      }
      
      circle(x + radius / 2, y - radius / 2, radius);
      
      x += radius + spacing;
    }
    
    y += radius + spacing;
  }
}
